%% Place a figure
% 1.) alt-caption (for lof)
% 2.) caption
% 3.) label
% 4.) figure placement (htbp)
% 5.) figure content
\newcommand{\FIG}[5][]{
  \begin{figure}[#4]
  \centering
  #5 
  \ifthenelse{\equal{#1}{}}{%
    \caption{#2}
  }{%
    \caption[#1]{#2}
  }
  \label{fig:#3}
  \end{figure}%
}
